package androidx.activity {

  public class ComponentActivity extends androidx.core.app.ComponentActivity implements androidx.lifecycle.LifecycleOwner {
    ctor public ComponentActivity();
    method public void addOnBackPressedHandler(androidx.lifecycle.LifecycleOwner, androidx.activity.OnBackPressedHandler);
    method public androidx.lifecycle.Lifecycle getLifecycle();
    method public void removeOnBackPressedHandler(androidx.activity.OnBackPressedHandler);
  }

  public abstract interface OnBackPressedHandler {
    method public abstract boolean onBackPressed();
  }

}
